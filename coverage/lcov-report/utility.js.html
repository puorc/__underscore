/**
 * Returns a random integer between min and max, inclusive. 
 * If you only pass one argument, it will return a number between 0 and that number.
 * @param {*} min 
 * @param {*} max 
 */
function random(min, max) {
    if (max == null) {
        max = min;
        min = 0;
    }
    return min + Math.floor(Math.random * (max - min + 1));
}
